"-- ============================================================\n-- MML Self-Healing Funnel System — Initial Schema\n-- ============================================================\n\n-- 1. FUNNELS\nCREATE TABLE funnels (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    name TEXT NOT NULL,\n    description TEXT,\n    offer_type TEXT NOT NULL CHECK (offer_type IN ('low_ticket', 'mid_ticket', 'high_ticket')),\n    offer_price_cents INTEGER NOT NULL,\n    status TEXT NOT NULL DEFAULT 'draft' CHECK (status IN ('draft', 'active', 'paused', 'archived')),\n    ghl_pipeline_id TEXT,\n    meta_campaign_ids TEXT[],\n    landing_page_slug TEXT UNIQUE,\n    created_at TIMESTAMPTZ DEFAULT now(),\n    updated_at TIMESTAMPTZ DEFAULT now()\n);\n\n-- 2. FUNNEL STAGES\nCREATE TABLE funnel_stages (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID NOT NULL REFERENCES funnels(id) ON DELETE CASCADE,\n    stage_type TEXT NOT NULL CHECK (stage_type IN (\n        'ad_impression', 'ad_click', 'page_view', 'page_conversion',\n        'email_sent', 'email_opened', 'email_clicked',\n        'sms_sent', 'sms_replied',\n        'ai_conversation', 'qualified', 'booking_made',\n        'booking_attended', 'proposal_sent', 'closed_won', 'closed_lost'\n    )),\n    stage_order INTEGER NOT NULL,\n    stage_name TEXT NOT NULL,\n    ghl_resource_id TEXT,\n    meta_adset_id TEXT,\n    created_at TIMESTAMPTZ DEFAULT now(),\n    UNIQUE(funnel_id, stage_order)\n);\n\n-- 3. STAGE BASELINES\nCREATE TABLE stage_baselines (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_stage_id UUID NOT NULL REFERENCES funnel_stages(id) ON DELETE CASCADE,\n    metric_name TEXT NOT NULL,\n    baseline_value NUMERIC NOT NULL,\n    warning_threshold NUMERIC NOT NULL,\n    critical_threshold NUMERIC NOT NULL,\n    lookback_window_hours INTEGER NOT NULL DEFAULT 24,\n    minimum_sample_size INTEGER NOT NULL DEFAULT 50,\n    updated_at TIMESTAMPTZ DEFAULT now(),\n    UNIQUE(funnel_stage_id, metric_name)\n);\n\n-- 4. CONTACTS\nCREATE TABLE contacts (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    ghl_contact_id TEXT UNIQUE,\n    email TEXT,\n    phone TEXT,\n    first_name TEXT,\n    last_name TEXT,\n    source TEXT,\n    source_campaign_id TEXT,\n    source_adset_id TEXT,\n    source_ad_id TEXT,\n    utm_source TEXT,\n    utm_medium TEXT,\n    utm_campaign TEXT,\n    utm_content TEXT,\n    utm_term TEXT,\n    lead_score INTEGER DEFAULT 0,\n    qualification_status TEXT DEFAULT 'unknown' CHECK (qualification_status IN (\n        'unknown', 'unqualified', 'nurturing', 'qualified', 'booked', 'closed_won', 'closed_lost'\n    )),\n    offer_type TEXT,\n    funnel_id UUID REFERENCES funnels(id),\n    tags TEXT[],\n    custom_fields JSONB DEFAULT '{}',\n    ghl_synced_at TIMESTAMPTZ,\n    created_at TIMESTAMPTZ DEFAULT now(),\n    updated_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_contacts_email ON contacts(email);\nCREATE INDEX idx_contacts_phone ON contacts(phone);\nCREATE INDEX idx_contacts_ghl_id ON contacts(ghl_contact_id);\nCREATE INDEX idx_contacts_qualification ON contacts(qualification_status);\nCREATE INDEX idx_contacts_funnel ON contacts(funnel_id);\n\n-- 5. PIPELINE EVENTS\nCREATE TABLE pipeline_events (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    contact_id UUID NOT NULL REFERENCES contacts(id) ON DELETE CASCADE,\n    funnel_id UUID REFERENCES funnels(id),\n    funnel_stage_id UUID REFERENCES funnel_stages(id),\n    event_type TEXT NOT NULL,\n    event_data JSONB DEFAULT '{}',\n    ghl_event_id TEXT,\n    occurred_at TIMESTAMPTZ NOT NULL DEFAULT now(),\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_pipeline_events_contact ON pipeline_events(contact_id, occurred_at DESC);\nCREATE INDEX idx_pipeline_events_funnel_stage ON pipeline_events(funnel_stage_id, occurred_at DESC);\nCREATE INDEX idx_pipeline_events_type ON pipeline_events(event_type);\n\n-- 6. METRIC SNAPSHOTS\nCREATE TABLE metric_snapshots (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_stage_id UUID NOT NULL REFERENCES funnel_stages(id),\n    snapshot_time TIMESTAMPTZ NOT NULL DEFAULT now(),\n    period_start TIMESTAMPTZ NOT NULL,\n    period_end TIMESTAMPTZ NOT NULL,\n    metric_name TEXT NOT NULL,\n    metric_value NUMERIC NOT NULL,\n    sample_size INTEGER NOT NULL,\n    source TEXT NOT NULL CHECK (source IN ('meta_ads', 'ghl', 'internal', 'resend', 'twilio', 'stripe')),\n    raw_payload JSONB,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_metric_snapshots_stage_time ON metric_snapshots(funnel_stage_id, snapshot_time DESC);\nCREATE INDEX idx_metric_snapshots_name_time ON metric_snapshots(metric_name, snapshot_time DESC);\n\n-- 7. EMAIL METRICS\nCREATE TABLE email_metrics (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID REFERENCES funnels(id),\n    email_id TEXT NOT NULL,\n    email_subject TEXT,\n    variant_id UUID,\n    period_start TIMESTAMPTZ NOT NULL,\n    period_end TIMESTAMPTZ NOT NULL,\n    sent_count INTEGER NOT NULL DEFAULT 0,\n    delivered_count INTEGER NOT NULL DEFAULT 0,\n    opened_count INTEGER NOT NULL DEFAULT 0,\n    clicked_count INTEGER NOT NULL DEFAULT 0,\n    replied_count INTEGER NOT NULL DEFAULT 0,\n    bounced_count INTEGER NOT NULL DEFAULT 0,\n    unsubscribed_count INTEGER NOT NULL DEFAULT 0,\n    open_rate NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN delivered_count > 0 THEN opened_count::NUMERIC / delivered_count ELSE 0 END\n    ) STORED,\n    click_rate NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN delivered_count > 0 THEN clicked_count::NUMERIC / delivered_count ELSE 0 END\n    ) STORED,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_email_metrics_funnel ON email_metrics(funnel_id);\n\n-- 8. AD METRICS\nCREATE TABLE ad_metrics (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID REFERENCES funnels(id),\n    meta_campaign_id TEXT NOT NULL,\n    meta_adset_id TEXT,\n    meta_ad_id TEXT,\n    period_start TIMESTAMPTZ NOT NULL,\n    period_end TIMESTAMPTZ NOT NULL,\n    impressions INTEGER NOT NULL DEFAULT 0,\n    clicks INTEGER NOT NULL DEFAULT 0,\n    spend_cents INTEGER NOT NULL DEFAULT 0,\n    conversions INTEGER NOT NULL DEFAULT 0,\n    leads INTEGER NOT NULL DEFAULT 0,\n    ctr NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN impressions > 0 THEN clicks::NUMERIC / impressions ELSE 0 END\n    ) STORED,\n    cpc_cents NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN clicks > 0 THEN spend_cents::NUMERIC / clicks ELSE 0 END\n    ) STORED,\n    cpl_cents NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN leads > 0 THEN spend_cents::NUMERIC / leads ELSE 0 END\n    ) STORED,\n    frequency NUMERIC,\n    relevance_score NUMERIC,\n    raw_payload JSONB,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_ad_metrics_campaign_time ON ad_metrics(meta_campaign_id, period_start DESC);\nCREATE INDEX idx_ad_metrics_funnel ON ad_metrics(funnel_id);\n\n-- 9. PAGE ANALYTICS\nCREATE TABLE page_analytics (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID REFERENCES funnels(id),\n    page_slug TEXT NOT NULL,\n    variant_id UUID,\n    period_start TIMESTAMPTZ NOT NULL,\n    period_end TIMESTAMPTZ NOT NULL,\n    visits INTEGER NOT NULL DEFAULT 0,\n    unique_visitors INTEGER NOT NULL DEFAULT 0,\n    conversions INTEGER NOT NULL DEFAULT 0,\n    avg_time_on_page_seconds NUMERIC,\n    avg_scroll_depth NUMERIC,\n    bounce_rate NUMERIC,\n    conversion_rate NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN visits > 0 THEN conversions::NUMERIC / visits ELSE 0 END\n    ) STORED,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_page_analytics_slug ON page_analytics(page_slug, period_start DESC);\n\n-- 10. A/B TESTS\nCREATE TABLE ab_tests (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID NOT NULL REFERENCES funnels(id),\n    funnel_stage_id UUID NOT NULL REFERENCES funnel_stages(id),\n    test_name TEXT NOT NULL,\n    test_type TEXT NOT NULL CHECK (test_type IN (\n        'email_subject', 'email_body', 'landing_page', 'cta', 'ad_creative', 'ad_copy', 'headline'\n    )),\n    status TEXT NOT NULL DEFAULT 'draft' CHECK (status IN (\n        'draft', 'running', 'paused', 'completed', 'cancelled'\n    )),\n    winner_variant_id UUID,\n    min_sample_per_variant INTEGER NOT NULL DEFAULT 100,\n    confidence_threshold NUMERIC NOT NULL DEFAULT 0.95,\n    started_at TIMESTAMPTZ,\n    completed_at TIMESTAMPTZ,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\n-- 11. A/B TEST VARIANTS\nCREATE TABLE ab_test_variants (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    ab_test_id UUID NOT NULL REFERENCES ab_tests(id) ON DELETE CASCADE,\n    variant_label TEXT NOT NULL,\n    variant_content JSONB NOT NULL,\n    is_control BOOLEAN DEFAULT false,\n    traffic_percentage NUMERIC NOT NULL DEFAULT 50,\n    impressions INTEGER NOT NULL DEFAULT 0,\n    conversions INTEGER NOT NULL DEFAULT 0,\n    conversion_rate NUMERIC GENERATED ALWAYS AS (\n        CASE WHEN impressions > 0 THEN conversions::NUMERIC / impressions ELSE 0 END\n    ) STORED,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\n-- Add FK from email_metrics to ab_test_variants\nALTER TABLE email_metrics ADD CONSTRAINT fk_email_variant\n    FOREIGN KEY (variant_id) REFERENCES ab_test_variants(id);\n\n-- 12. OPTIMIZATION ACTIONS (the audit trail)\nCREATE TABLE optimization_actions (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID NOT NULL REFERENCES funnels(id),\n    funnel_stage_id UUID REFERENCES funnel_stages(id),\n    trigger_snapshot_id UUID REFERENCES metric_snapshots(id),\n    action_type TEXT NOT NULL CHECK (action_type IN (\n        'swap_email_subject', 'swap_email_body', 'adjust_ad_budget',\n        'adjust_ad_targeting', 'change_cta', 'swap_landing_page',\n        'modify_offer', 'pause_ad', 'launch_ad_variant',\n        'change_email_timing', 'trigger_sms_sequence',\n        'adjust_qualification_criteria', 'escalate_to_human',\n        'revert_previous_action', 'custom'\n    )),\n    risk_tier TEXT NOT NULL DEFAULT 'low' CHECK (risk_tier IN ('low', 'medium', 'high')),\n    status TEXT NOT NULL DEFAULT 'proposed' CHECK (status IN (\n        'proposed', 'approved', 'executing', 'executed',\n        'reviewing', 'kept', 'reverted', 'rejected', 'failed'\n    )),\n    diagnosis TEXT NOT NULL,\n    action_details JSONB NOT NULL,\n    previous_state JSONB,\n    executed_at TIMESTAMPTZ,\n    review_at TIMESTAMPTZ,\n    review_window_hours INTEGER DEFAULT 48,\n    impact_assessment JSONB,\n    impact_verdict TEXT CHECK (impact_verdict IN ('positive', 'neutral', 'negative')),\n    executed_by TEXT DEFAULT 'system',\n    approved_by TEXT,\n    created_at TIMESTAMPTZ DEFAULT now(),\n    updated_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_optimization_actions_status ON optimization_actions(status);\nCREATE INDEX idx_optimization_actions_review ON optimization_actions(review_at) WHERE status = 'executed';\nCREATE INDEX idx_optimization_actions_funnel ON optimization_actions(funnel_id);\n\n-- 13. AI CONVERSATIONS\nCREATE TABLE conversations (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    contact_id UUID NOT NULL REFERENCES contacts(id) ON DELETE CASCADE,\n    funnel_id UUID REFERENCES funnels(id),\n    channel TEXT NOT NULL CHECK (channel IN ('sms', 'email', 'chat')),\n    conversation_type TEXT NOT NULL DEFAULT 'qualification' CHECK (conversation_type IN (\n        'qualification', 'follow_up', 'no_show_recovery', 'reengagement', 'support'\n    )),\n    status TEXT NOT NULL DEFAULT 'active' CHECK (status IN (\n        'active', 'qualified', 'disqualified', 'booked', 'escalated', 'closed'\n    )),\n    messages JSONB NOT NULL DEFAULT '[]',\n    qualification_signals JSONB DEFAULT '{}',\n    message_count INTEGER NOT NULL DEFAULT 0,\n    last_message_at TIMESTAMPTZ,\n    created_at TIMESTAMPTZ DEFAULT now(),\n    updated_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_conversations_contact ON conversations(contact_id);\nCREATE INDEX idx_conversations_status ON conversations(status);\n\n-- 14. OBJECTION PLAYBOOK\nCREATE TABLE objection_playbook (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    offer_type TEXT NOT NULL,\n    objection_category TEXT NOT NULL CHECK (objection_category IN (\n        'price', 'timing', 'authority', 'trust', 'need', 'competitor', 'generic'\n    )),\n    objection_pattern TEXT NOT NULL,\n    response_strategy TEXT NOT NULL,\n    example_responses TEXT[],\n    effectiveness_score NUMERIC DEFAULT 0.5,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\n-- 15. ALERTS\nCREATE TABLE alerts (\n    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),\n    funnel_id UUID REFERENCES funnels(id),\n    optimization_action_id UUID REFERENCES optimization_actions(id),\n    severity TEXT NOT NULL CHECK (severity IN ('info', 'warning', 'critical')),\n    title TEXT NOT NULL,\n    message TEXT NOT NULL,\n    acknowledged BOOLEAN DEFAULT false,\n    acknowledged_at TIMESTAMPTZ,\n    created_at TIMESTAMPTZ DEFAULT now()\n);\n\nCREATE INDEX idx_alerts_unacknowledged ON alerts(created_at DESC) WHERE NOT acknowledged;\n\n-- ============================================================\n-- VIEWS\n-- ============================================================\n\n-- Funnel health at a glance\nCREATE VIEW v_funnel_health AS\nSELECT\n    f.id AS funnel_id,\n    f.name,\n    f.offer_type,\n    f.status,\n    COUNT(DISTINCT fs.id) AS total_stages,\n    COUNT(DISTINCT CASE\n        WHEN latest.metric_value < sb.critical_threshold AND latest.sample_size >= sb.minimum_sample_size\n        THEN fs.id\n    END) AS critical_stages,\n    COUNT(DISTINCT CASE\n        WHEN latest.metric_value < sb.warning_threshold\n        AND latest.metric_value >= sb.critical_threshold\n        AND latest.sample_size >= sb.minimum_sample_size\n        THEN fs.id\n    END) AS warning_stages,\n    CASE\n        WHEN COUNT(DISTINCT CASE\n            WHEN latest.metric_value < sb.critical_threshold AND latest.sample_size >= sb.minimum_sample_size\n            THEN fs.id END) > 0 THEN 'critical'\n        WHEN COUNT(DISTINCT CASE\n            WHEN latest.metric_value < sb.warning_threshold AND latest.sample_size >= sb.minimum_sample_size\n            THEN fs.id END) > 0 THEN 'warning'\n        ELSE 'healthy'\n    END AS health_status\nFROM funnels f\nJOIN funnel_stages fs ON fs.funnel_id = f.id\nLEFT JOIN stage_baselines sb ON sb.funnel_stage_id = fs.id AND sb.metric_name = 'conversion_rate'\nLEFT JOIN LATERAL (\n    SELECT metric_value, sample_size\n    FROM metric_snapshots\n    WHERE funnel_stage_id = fs.id AND metric_name = 'conversion_rate'\n    ORDER BY snapshot_time DESC\n    LIMIT 1\n) latest ON true\nWHERE f.status = 'active'\nGROUP BY f.id, f.name, f.offer_type, f.status;\n\n-- Stage-by-stage conversion funnel\nCREATE VIEW v_funnel_conversion AS\nSELECT\n    fs.funnel_id,\n    fs.stage_order,\n    fs.stage_name,\n    fs.stage_type,\n    latest.metric_value AS current_rate,\n    sb.baseline_value AS baseline_rate,\n    sb.warning_threshold,\n    sb.critical_threshold,\n    latest.sample_size,\n    latest.snapshot_time AS last_measured,\n    CASE\n        WHEN latest.metric_value IS NULL THEN 'no_data'\n        WHEN latest.sample_size < COALESCE(sb.minimum_sample_size, 50) THEN 'insufficient_data'\n        WHEN latest.metric_value < sb.critical_threshold THEN 'critical'\n        WHEN latest.metric_value < sb.warning_threshold THEN 'warning'\n        ELSE 'healthy'\n    END AS stage_health\nFROM funnel_stages fs\nLEFT JOIN stage_baselines sb ON sb.funnel_stage_id = fs.id AND sb.metric_name = 'conversion_rate'\nLEFT JOIN LATERAL (\n    SELECT metric_value, sample_size, snapshot_time\n    FROM metric_snapshots\n    WHERE funnel_stage_id = fs.id AND metric_name = 'conversion_rate'\n    ORDER BY snapshot_time DESC\n    LIMIT 1\n) latest ON true\nORDER BY fs.funnel_id, fs.stage_order;\n\n-- ============================================================\n-- ROW LEVEL SECURITY\n--
